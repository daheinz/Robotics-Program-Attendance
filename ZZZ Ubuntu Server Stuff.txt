# Get attendance records
SELECT 
  u.alias, 
  u.first_name, 
  u.last_name,
  a.check_in_time, 
  a.check_out_time, 
  a.duration_minutes
FROM attendance_sessions a
JOIN users u ON a.user_id = u.id
WHERE u.alias = 'their_alias'
ORDER BY a.check_in_time DESC;

#Update User role
UPDATE users SET role = 'mentor', updated_at = CURRENT_TIMESTAMP WHERE alias = 'their_alias';
